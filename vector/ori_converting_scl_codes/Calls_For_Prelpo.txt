
REGION DESCRIPTION
(/*
This is an example program to call the winder function block and operate the line axis
*/)
END_REGION DESCRIPTION

REGION FIRST CALL
  // Read cycle time of calling OB
  IF #Initial_Call THEN
    "LConSMC_Data".WinderInput["TOP_WINDER"].tCycle     := UDINT_TO_REAL(#cycleTime / 1000000);                 // Convert unit: [ns] --> [ms]
    "LConSMC_Data".WinderInput["BOTTOM_WINDER"].tCycle  := "LConSMC_Data".WinderInput["TOP_WINDER"].tCycle;
    "LConSMC_Data".sectionalInput[1].tCycle             := "LConSMC_Data".WinderInput["TOP_WINDER"].tCycle;
  END_IF;
END_REGION

REGION LEADING VALUE
  // Leading value interconnection
  // Copy to local temp variables
  // Line setpoint values
  #tempLinePosition := "line".Position * "LConSMC_Data".lengthConversion;               // Convert unit to [m]
  #tempLineVelocity := "line".Velocity * "LConSMC_Data".velocityConversion;             // Convert unit to [m/min]
  #tempLineAcceleration := "line".Acceleration * "LConSMC_Data".accelerationConversion; // Convert unit to [m/s^2]
  // Line actual values (used for diameter calculation)
  #tempLineActPos := "line".ActualPosition * "LConSMC_Data".lengthConversion;           // Convert unit to [m]
  #tempLineActVel := "line".ActualVelocity * "LConSMC_Data".velocityConversion;         // Convert unit to [m/min]
  
  REGION WINDER
    // Forward temporary variables to winder interface
    "LConSMC_Data".WinderInput["TOP_WINDER"].lineAxisMotionVector.s := #tempLinePosition;
    "LConSMC_Data".WinderInput["TOP_WINDER"].lineAxisMotionVector.v := #tempLineVelocity;
    "LConSMC_Data".WinderInput["TOP_WINDER"].lineAxisMotionVector.a := #tempLineAcceleration;
    
    "LConSMC_Data".WinderInput["TOP_WINDER"].lineAxisActualValues.s := #tempLineActPos;
    "LConSMC_Data".WinderInput["TOP_WINDER"].lineAxisActualValues.v := #tempLineActVel;
    
    "LConSMC_Data".WinderInput["BOTTOM_WINDER"].lineAxisMotionVector.s := #tempLinePosition;
    "LConSMC_Data".WinderInput["BOTTOM_WINDER"].lineAxisMotionVector.v := #tempLineVelocity;
    "LConSMC_Data".WinderInput["BOTTOM_WINDER"].lineAxisMotionVector.a := #tempLineAcceleration;
    
    "LConSMC_Data".WinderInput["BOTTOM_WINDER"].lineAxisActualValues.s := #tempLineActPos;
    "LConSMC_Data".WinderInput["BOTTOM_WINDER"].lineAxisActualValues.v := #tempLineActVel;
  END_REGION
  
  REGION SECTIONAL DRIVE
    "LConSMC_Data".sectionalInput[1].lineAxisMotionVector.s := #tempLinePosition;
    "LConSMC_Data".sectionalInput[1].lineAxisMotionVector.v := #tempLineVelocity;
    "LConSMC_Data".sectionalInput[1].lineAxisMotionVector.a := #tempLineAcceleration;
  END_REGION

END_REGION ;
